// Action script...

// [Action in Frame 1]
walkspeed = 5;
walkspeedback = 2;
Samuraiattack = False;
hpValue = 2;
HP1Value = 0.150000;
SamuraiChar.onEnterFrame = function ()
{
    if (Key.isDown(65) && !Samuraiattack)
    {
        this._x = this._x - walkspeedback;
        this.gotoAndPlay("Samurai1Walkback");
    }
    else if (Key.isDown(68) && !Samuraiattack)
    {
        this._x = this._x + walkspeed;
        this.gotoAndStop("Samurai1Walk");
    }
    else if (Key.isDown(71) && !Samuraiattack)
    {
        this.gotoAndPlay("Samurai1Attack");
    }
    else if (Key.isDown(72) && !Samuraiattack)
    {
        this.gotoAndPlay("Samurai1Special");
    }
    else if (Key.isDown(84) && !Samuraiattack)
    {
        this.gotoAndPlay("Samurai1Attack2");
    }
    else if (!Samuraiattack)
    {
        this.gotoAndStop("Samurai1Stand");
    } // end else if
    if (SamuraiChar.Samurai1Attack.Pad.hitTest(this) && !Samurai2Hit)
    {
        this.gotoAndStop("Samurai2Hit");
        if (hp2._xscale > 0)
        {
            hp2._xscale = hp2._xscale - hpValue;
        }
        else if (SamuraiChar.Samurai1Special.hitTest(this) && !Samurai2Hit)
        {
            hp2._xscale = hp2._xscale - hpValue;
        } // end else if
    }
    else if (SamuraiChar.Samurai2Special.hitTest(this) && !Samurai2Hit)
    {
        this.gotoAndStop("Samurai2Hit");
    }
    else if (!Samurai2Hit)
    {
        this.gotoAndStop("Samurai2Stand");
    } // end else if
};
SamuraiChar2.onEnterFrame = function ()
{
    if (Key.isDown(37) && !Samurai2attack)
    {
        this._x = this._x - walkspeed;
        this.gotoAndPlay("Samurai2Walk");
    }
    else if (Key.isDown(39) && !Samurai2attack)
    {
        this._x = this._x + walkspeedback;
        this.gotoAndPlay("Samurai2Walk");
    }
    else if (Key.isDown(77))
    {
        this.gotoAndPlay("Samurai2Attack");
    }
    else if (!Samurai2attack)
    {
        this.gotoAndPlay("Samurai2Stand");
    } // end else if
    if (SamuraiChar2.Samurai2Attack.hitTest(this) && !Samurai1Hit)
    {
        if (HP1._xscale > 0)
        {
            HP1._xscale = HP1._xscale - HP1Value;
        }
        else if (HP1._xscale = 0)
        {
            Close();
        } // end if
    } // end else if
};
